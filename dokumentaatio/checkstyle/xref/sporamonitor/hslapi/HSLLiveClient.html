3   import com.google.gson.*;
4   import com.google.gson.reflect.TypeToken;
5   import org.asynchttpclient.AsyncHttpClient;
6   import org.asynchttpclient.Dsl;
7   
8   import java.io.IOException;
9   import java.lang.reflect.Type;
10  import java.nio.charset.Charset;
11  import java.util.List;
12  import java.util.concurrent.CompletableFuture;
13  import java.util.function.Supplier;
14  
15  /**
16   * An API client to HSL's Live API (http://dev.hsl.fi/)
17   */
18  public class HSLLiveClient {
19      private static final String ENDPOINT_URI = "http://dev.hsl.fi/siriaccess/vm/json?operatorRef=HSL";
20  
21      private Gson gson;
22      private AsyncHttpClient httpClient;
23  
24      public HSLLiveClient() {
25          this(Dsl::asyncHttpClient);
26      }
27  
28      public HSLLiveClient(Supplier<AsyncHttpClient> httpClientSupplier) {
29          this.httpClient = httpClientSupplier.get();
30      }
31  
32      /**
33       * Returns the list of vehicles currently in transit by HSL
34       * @returns a future containing a list of vehicles
35       * @throws IOException
36       */
37      public CompletableFuture<List<Vehicle>> vehicles() throws IOException {
38          return httpClient
39                  .prepareGet(ENDPOINT_URI)
40                  .execute()
41                  .toCompletableFuture()
42                  .thenApply(resp -> resp.getResponseBody(Charset.forName("UTF-8")))
43                  .thenApply(this::parseSiriJson);
44      }
45  
46      private List<Vehicle> parseSiriJson(String json) {
47          JsonParser parser = new JsonParser();
48          JsonElement container = parser.parse(json);
49          JsonArray vehicleActivity = container
50                  .getAsJsonObject()
51                  .getAsJsonObject("Siri")
52                  .getAsJsonObject("ServiceDelivery")
53                  .getAsJsonArray("VehicleMonitoringDelivery")
54                  .get(0).getAsJsonObject()
55                  .getAsJsonArray("VehicleActivity");
56          Type listType = new TypeToken<List<Vehicle>>() { }.getType();
57          return buildGson().fromJson(vehicleActivity, listType);
58      }
59  
60      private Gson buildGson() {
61          if (gson != null) {
62              return gson;
63          }
64          GsonBuilder builder = new GsonBuilder();
65          builder.registerTypeAdapter(Vehicle.class, new VehicleDeserializer());
66          gson = builder.create();
67          return gson;
68      }
69  }
