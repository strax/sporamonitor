3   import java.util.function.Consumer;
4   
5   /**
6    * A type-safe implementation of the subject in the observer pattern that uses Java 8 lambdas for observers.
7    * @param <T> the type of the state of the subject
8    */
9   public interface Subject<T> {
10      /**
11       * Registers a new observer to be notified when the subject's state changes.
12       * @param observer the observer to add to this subject
13       */
14      void subscribe(Consumer<T> observer);
15  
16      /**
17       * Detaches a previously registered observer from this subject.
18       * @param observer the observer to remove from this subject
19       * @throws IllegalArgumentException if the observer is not attached to the subject
20       */
21      void unsubscribe(Consumer<T> observer) throws IllegalArgumentException;
22  }
